CREATE PROCEDURE procCustomerSave
	@ID int = NULL,
	@Color VARCHAR(20)
AS
  DECLARE @ExistColor INT
  -- IF THE INDEX PARAMETER IS NULL THEN NEW ITEM IS TO BE INSERTED, OTHERWISE UPDATE EXISTING ITEM
  IF (@ID IS NULL)
	  BEGIN
		  BEGIN
			  INSERT INTO Color(ColorName)
			  VALUES (@Color)
			  SELECT @ExistColor = SCOPE_IDENTITY()
		  END
	  END
	  RETURN @ExistColor
GO
